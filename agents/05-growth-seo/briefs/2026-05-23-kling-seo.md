### B-3. Next.js App Router 구현 방법

#### `src/app/sitemap.ts` (Next.js 15 App Router)

```typescript
import { MetadataRoute } from 'next'
import { createClient } from '@/lib/supabase/server'

export const revalidate = 3600 // 1시간 캐시

export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
  const baseUrl = 'https://aiarklive.com'
  const supabase = await createClient()

  // 정적 라우트
  const staticRoutes: MetadataRoute.Sitemap = [
    { url: baseUrl, lastModified: new Date(), changeFrequency: 'daily', priority: 1.0 },
    { url: `${baseUrl}/explore`, lastModified: new Date(), changeFrequency: 'daily', priority: 0.9 },
    { url: `${baseUrl}/discover`, lastModified: new Date(), changeFrequency: 'weekly', priority: 0.9 },
    { url: `${baseUrl}/leaderboard`, lastModified: new Date(), changeFrequency: 'daily', priority: 0.8 },
    { url: `${baseUrl}/challenges`, lastModified: new Date(), changeFrequency: 'weekly', priority: 0.8 },
    { url: `${baseUrl}/collections`, lastModified: new Date(), changeFrequency: 'weekly', priority: 0.6 },
    { url: `${baseUrl}/tools/kling`, lastModified: new Date(), changeFrequency: 'weekly', priority: 0.9 },
  ]

  // 동적: public 영상 (최대 1000개, 최신순)
  const { data: videos } = await supabase
    .from('videos')
    .select('id, updated_at')
    .order('created_at', { ascending: false })
    .limit(1000)

  const videoRoutes: MetadataRoute.Sitemap = (videos ?? []).map((v) => ({
    url: `${baseUrl}/video/${v.id}`,
    lastModified: new Date(v.updated_at),
    changeFrequency: 'monthly',
    priority: 0.7,
  }))

  // 동적: public 프로필
  const { data: profiles } = await supabase
    .from('users')
    .select('id, updated_at')
    .limit(500)

  const profileRoutes: MetadataRoute.Sitemap = (profiles ?? []).map((p) => ({
    url: `${baseUrl}/profile/${p.id}`,
    lastModified: new Date(p.updated_at),
    changeFrequency: 'monthly',
    priority: 0.5,
  }))

  return [...staticRoutes, ...videoRoutes, ...profileRoutes]
}
```

> **Cloudflare Pages 주의:** edge runtime에서 `createClient`가 cookies()를 사용하므로  
> `sitemap.ts`는 Node.js runtime으로 실행해야 함. 파일 상단에:  
> ```typescript
> export const runtime = 'nodejs'
> ```

#### `src/app/robots.ts`

### C-2. video/[id]/page.tsx metadata 연동

`generateMetadata`에서 OG 태그를 명시적으로 설정해야 소셜 미디어 크롤러가 올바르게 읽음:

```typescript
export async function generateMetadata({ params }: { params: { id: string } }): Promise<Metadata> {
  const supabase = await createClient()
  const { data: video } = await supabase
    .from('videos')
    .select('title, description, thumbnail_url, ai_tool')
    .eq('id', params.id)
    .single()

  if (!video) return { title: 'Video | AIARKLIVE' }

  const ogImageUrl = `https://aiarklive.com/video/${params.id}/opengraph-image`

  return {
    title: `${video.title} — ${video.ai_tool ?? 'AI Video'} | AIARKLIVE`,
    description: video.description ?? `Watch this AI-generated video made with ${video.ai_tool} on AIARKLIVE.`,
    openGraph: {
      title: video.title,
      description: video.description ?? '',
      images: [{ url: ogImageUrl, width: 1200, height: 630, alt: video.title }],
      type: 'video.other',
      siteName: 'AIARKLIVE',
    },
    twitter: {
      card: 'summary_large_image',
      title: video.title,
      description: video.description ?? '',
      images: [ogImageUrl],
    },
  }
}
```
